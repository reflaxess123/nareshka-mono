// Оригинальная функция с коллбэком, которая выполняется с задержкой  
function delayedCallback(data, delay, callback) {  
    setTimeout(() => {  
        if (typeof data === 'undefined') {  
            callback("No data provided", null);  
        } else {  
            callback(null, `Processed: ${data}`);  
        }  
    }, delay);  
}
