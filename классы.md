# tbq
почему наружен принцип инверсии зависимости

```js

// Функция, которая выводит сообщение в консоль
function do_work(): void {
  console.log('I am working');
}
// Класс User с полями name и age
class User {
  name: string;
  age: number;
  // Конструктор, который инициализирует поля name и age
  constructor(name: string, age: number, do_work: () => void) {
    this.name = name;
    this.age = age;
    this.do_work = do_work
  }
  // Метод do_work, который вызывает функцию do_work
  do_work(): void {
    this.do_work();
  }
}
// Создаем новый экземпляр класса User с именем 'Anton' и возрастом 37
let user = new User('Anton', 37);
// Вызываем метод do_work у экземпляра класса User
user.do_work();

```
